64void CmrModule::initialize(const string &modname)
65{
66 BESDebug::Register(MODULE);
67 BESDEBUG(MODULE, prolog << "BEGIN (modname: " << modname << ")" << endl);
68 if (!BESCatalogList::TheCatalogList()->ref_catalog(CMR_CATALOG_NAME)) {
69 BESDEBUG(MODULE, prolog << "Enabling Cmr Catalog Module." << endl);
70 BESCatalogList::TheCatalogList()->add_catalog(new cmr::CmrCatalog(CMR_CATALOG_NAME));
71 }
72 else {
73 BESDEBUG(MODULE, prolog << "CMR Catalog Module is not enabled." << endl);
74 }
75
76 if (!BESContainerStorageList::TheList()->ref_persistence(CMR_CATALOG_NAME) &&
77 TheBESKeys::TheKeys()->read_bool_key("CMR.UseContainerAccess", false) ){
78 BESDEBUG(MODULE, prolog << "Enabling CmrContainerStorage for access." << endl);
79 BESContainerStorageList::TheList()->add_persistence(new cmr::CmrContainerStorage(CMR_CATALOG_NAME));
80 }
81 else {
82 BESDEBUG(MODULE, prolog << "CmrContainerStorage is not enabled." << endl);
83 }
84
85 BESDEBUG(MODULE, "Done Initializing CMR Handler: " << modname << endl);
86}
