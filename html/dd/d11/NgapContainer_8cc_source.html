86void NgapContainer::set_real_name_using_cmr_or_cache()
87{
88#ifndef NDEBUG
89 BESStopWatch besTimer;
90 if (BESISDEBUG(MODULE) || BESISDEBUG(TIMING_LOG_KEY) || BESLog::TheLog()->is_verbose()) {
91 besTimer.start(prolog + get_real_name());
92 }
93#endif
94
95 bool found;
96 string uid = BESContextManager::TheManager()->get_context(EDL_UID_KEY, found);
97 BESDEBUG(MODULE, prolog << "EDL_UID_KEY(" << EDL_UID_KEY << "): " << uid << endl);
98
99 // If using the cache, look there. Note that the UID is part of the key to the cached data.
100 string url_key = d_ngap_path + ':' + uid;
101 string real_name;
102 if (NgapRequestHandler::d_use_cmr_cache) {
103 if (NgapRequestHandler::d_cmr_mem_cache.get(url_key, real_name)) {
104 set_real_name(real_name);
105 set_relative_name(real_name);
106 CACHE_LOG(prolog + "CMR Cache hit, translated URL: " + get_real_name() + '\n');
107 BESDEBUG(MODULE, prolog << "END (obj_addr: " << (void *) this << ")" << endl);
108 return;
109 }
110 else {
111 CACHE_LOG(prolog + "CMR Cache miss, URL: " + get_real_name() + '\n');
112 }
113 }
114
115 real_name = NgapApi::convert_ngap_resty_path_to_data_access_url(get_real_name());
116 set_real_name(real_name);
117
118 // Because we know the name is really a URL, then we know the "relative_name" is meaningless
119 // So we set it to be the same as "name"
120 set_relative_name(real_name);
121
122 // If using the CMR cache, cache the response.
123 if (NgapRequestHandler::d_use_cmr_cache) {
124 NgapRequestHandler::d_cmr_mem_cache.put(url_key, real_name);
125 CACHE_LOG(prolog + "CMR Cache put, translated URL: " + get_real_name() + '\n');
126 }
127}

155bool
156NgapContainer::get_content_filters(map<string, string, std::less<>> &content_filters) const
157{
158 if (inject_data_url()) {
159 const string data_access_url_str = get_real_name();
160 const string missing_data_url_str = data_access_url_str + ".missing";
161 const string href=R"(href=")";
162 const string trusted_url_hack=R"(" dmrpp:trust="true")";
163 const string data_access_url_key = href + DATA_ACCESS_URL_KEY + "\"";
164 const string data_access_url_with_trusted_attr_str = href + data_access_url_str + trusted_url_hack;
165 const string missing_data_access_url_key = href + MISSING_DATA_ACCESS_URL_KEY + "\"";
166 const string missing_data_url_with_trusted_attr_str = href + missing_data_url_str + trusted_url_hack;
167
168 content_filters.clear();
169 content_filters.insert(pair<string,string>(data_access_url_key, data_access_url_with_trusted_attr_str));
170 content_filters.insert(pair<string,string>(missing_data_access_url_key, missing_data_url_with_trusted_attr_str));
171 return true;
172 }
173
174 return false;
175}

177static bool file_to_string(int fd, string &content) {
178 vector<char> buffer(4096);
179 ssize_t bytes_read;
180 while ((bytes_read = read(fd, buffer.data(), buffer.size())) > 0) {
181 content.append(buffer.data(), bytes_read);
182 }
183 return bytes_read == 0;
184}

205bool NgapContainer::get_dmrpp_from_cache_or_remote_source(string &dmrpp_string) const {
206#ifndef NDEBUG
207 BESStopWatch besTimer;
208 if (BESISDEBUG(MODULE) || BESISDEBUG(TIMING_LOG_KEY) || BESLog::TheLog()->is_verbose()) {
209 besTimer.start("NGAP Container access: " + get_real_name());
210 }
211#endif
212
213 if (NgapRequestHandler::d_use_dmrpp_cache) {
214 if (NgapRequestHandler::d_dmrpp_mem_cache.get(get_real_name(), dmrpp_string)) {
215 // set_container_type() because access() is called from both the framework and the DMR++ handler
216 CACHE_LOG(prolog + "Memory Cache hit, DMR++: " + get_real_name() + '\n');
217 return true;
218 }
219 else {
220 CACHE_LOG(prolog + "Memory Cache miss, DMR++: " + get_real_name() + '\n');
221 }
222 }
223
224 // Before going over the network to get the DMR++, look in the FileCache.
225 // If found, put it in the memory cache and return it as a string.
226 if (NgapRequestHandler::d_use_dmrpp_cache) {
227 FileCache::Item item;
228 if (NgapRequestHandler::d_dmrpp_file_cache.get(FileCache::hash_key(get_real_name()), item)) { // got it
229 // read data from the file into the string.
230 CACHE_LOG(prolog + "File Cache hit, DMR++: " + get_real_name() + '\n');
231 if (file_to_string(item.get_fd(), dmrpp_string)) {
232 // put it in the memory cache
233 NgapRequestHandler::d_dmrpp_mem_cache.put(get_real_name(), dmrpp_string);
234 CACHE_LOG(prolog + "Memory Cache put, DMR++: " + get_real_name() + '\n');
235 return true;
236 }
237 else {
238 ERROR_LOG("NgapContainer::access() - failed to read DMR++ from file cache\n");
239 return false;
240 }
241 }
242 else {
243 CACHE_LOG(prolog + "File Cache miss, DMR++: " + get_real_name() + '\n');
244 }
245 } // end of FileCache::Item scope
246
247 string dmrpp_url_str = get_real_name() + ".dmrpp"; // The URL of the DMR++ file
248
249 // Else, the DMR++ is neither in the memory cache nor the file cache.
250 // Read it from S3, etc., and filter it. Put it in the memory cache.
251#ifndef NDEBUG
252 BESStopWatch besTimer2;
253 if (BESISDEBUG(MODULE) || BESISDEBUG(TIMING_LOG_KEY) || BESLog::TheLog()->is_verbose()) {
254 besTimer2.start("DMR++ retrieval: " + dmrpp_url_str);
255 }
256#endif
257
258 // This code throws an exception if there is a problem. jhrg 11/16/23
259 curl::http_get(dmrpp_url_str, dmrpp_string);
260#if 0
261 vector<char> buffer;
262 curl::http_get(dmrpp_url_str, buffer);
263 copy(buffer.begin(), buffer.end(), back_inserter(dmrpp_string));
264 buffer.clear(); // keep the original for as little time as possible.
265#endif
266
267 map<string,string, std::less<>> content_filters;
268 if (get_content_filters(content_filters)) {
269 filter_response(content_filters, dmrpp_string);
270 }
271
272 // if we get here, the DMR++ has been pulled over the network. Put it in both caches.
273 // The memory cache is for use by this process, the file cache for other processes/VMs
274 if (NgapRequestHandler::d_use_dmrpp_cache) {
275 NgapRequestHandler::d_dmrpp_mem_cache.put(get_real_name(), dmrpp_string);
276 CACHE_LOG(prolog + "Memory Cache put, DMR++: " + get_real_name() + '\n');
277
278 FileCache::PutItem item(NgapRequestHandler::d_dmrpp_file_cache);
279 if (NgapRequestHandler::d_dmrpp_file_cache.put(FileCache::hash_key(get_real_name()), item)) {
280 // Do this in a child thread someday. jhrg 11/14/23
281 if (write(item.get_fd(), dmrpp_string.data(), dmrpp_string.size()) != dmrpp_string.size()) {
282 ERROR_LOG("NgapContainer::access() - failed to write DMR++ to file cache\n");
283 return false;
284 }
285 CACHE_LOG(prolog + "File Cache put, DMR++: " + get_real_name() + '\n');
286 }
287 else {
288 ERROR_LOG("NgapContainer::access() - failed to put DMR++ in file cache\n");
289 return false;
290 }
291
292 if (!NgapRequestHandler::d_dmrpp_file_cache.purge()) {
293 ERROR_LOG("NgapContainer::access() - call to FileCache::purge() failed\n");
294 }
295 }
296
297 return true;
298}

310string NgapContainer::access() {
311 set_real_name_using_cmr_or_cache();
312
313 string dmrpp_string;
314 get_dmrpp_from_cache_or_remote_source(dmrpp_string);
315
316 set_attributes("as-string"); // This means access() returns a string. jhrg 10/19/23
317 // Originally, this was either hard-coded (as it is now) or was set using the 'extension'
318 // on the URL. But it's always a DMR++. jhrg 11/16/23
319 set_container_type("dmrpp");
320 return dmrpp_string;
321}
