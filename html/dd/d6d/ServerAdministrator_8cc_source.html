186std::string ServerAdministrator::get(const string &key){
187 string lkey = BESUtil::lowercase(key);
188 auto result = d_admin_info.find(lkey);
189 if(result == d_admin_info.end()){
190 return "";
191 }
192 return result->second;
193}
