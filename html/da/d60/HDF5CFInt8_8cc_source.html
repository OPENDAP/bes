57bool HDF5CFInt8::read()
58{
59 BESDEBUG("h5","Coming to HDF5CFInt8 read "<<endl);
60
61 if (read_p())
62 return true;
63
64 hid_t file_id = H5Fopen(filename.c_str(),H5F_ACC_RDONLY,H5P_DEFAULT);
65 if(file_id < 0) {
66 throw InternalErr(__FILE__,__LINE__, "Fail to obtain the HDF5 file ID .");
67 }
68
69 hid_t dset_id = -1;
70 dset_id = H5Dopen2(file_id,dataset().c_str(),H5P_DEFAULT);
71 if(dset_id < 0) {
72 H5Fclose(file_id);
73 throw InternalErr(__FILE__,__LINE__, "Fail to obtain the dataset .");
74 }
75
76 hid_t dtypeid = H5Dget_type(dset_id);
77 if(dtypeid < 0) {
78 H5Dclose(dset_id);
79 H5Fclose(file_id);
80 throw InternalErr(__FILE__,__LINE__, "Fail to obtain the datatype .");
81 }
82
83 hid_t memtype = H5Tget_native_type(dtypeid, H5T_DIR_ASCEND);
84 if (memtype < 0){
85 H5Tclose(dtypeid);
86 H5Dclose(dset_id);
87 H5Fclose(file_id);
88 throw InternalErr(__FILE__, __LINE__, "Cannot obtain the native datatype.");
89 }
90
91 try {
92
93 dods_int8 buf;
94 get_data(dset_id, (void *) &buf);
95
96 set_read_p(true);
97 set_value(buf);
98
99 if(H5Tclose(memtype) < 0) {
100 throw InternalErr(__FILE__, __LINE__, "Unable to close the memory datatype.");
101 }
102 if(H5Tclose(dtypeid) < 0) {
103 throw InternalErr(__FILE__, __LINE__, "Unable to close the datatype id.");
104 }
105 // Release the handles.
106 if (H5Dclose(dset_id) < 0) {
107 throw InternalErr(__FILE__, __LINE__, "Unable to close the dset.");
108 }
109 H5Fclose(file_id);
110 }
111 catch(...) {
112 H5Tclose(memtype);
113 H5Tclose(dtypeid);
114 H5Dclose(dset_id);
115 H5Fclose(file_id);
116 throw;
117 }
118
119 return true;
120
121}
