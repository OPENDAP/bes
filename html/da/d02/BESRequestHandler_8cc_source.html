112string BESRequestHandler::get_method_names()
113{
114 string ret = "";
115 bool first_name = true;
116 BESRequestHandler::Handler_citer i = _handler_list.begin();
117 for (; i != _handler_list.end(); i++) {
118 if (!first_name) ret += ", ";
119 ret += (*i).first;
120 first_name = false;
121 }
122 return ret;
