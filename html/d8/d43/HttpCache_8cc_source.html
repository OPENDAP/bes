104HttpCache *
105HttpCache::get_instance() {
106 if (d_instance == nullptr) {
107 static std::once_flag d_euc_init_once;
108 std::call_once(d_euc_init_once, []() {
109 d_instance.reset(new HttpCache());
110 });
111
112 string cacheDir = get_http_cache_dir_from_config();
113 string cachePrefix = get_http_cache_prefix_from_config();
114 unsigned long cacheSizeMbytes = get_http_cache_size_from_config();
115
116 BESDEBUG(HTTP_MODULE, prolog << "Cache configuration params: " << cacheDir << ", " << cachePrefix << ", "
117 << cacheSizeMbytes << endl);
118 d_instance->initialize(cacheDir, cachePrefix, cacheSizeMbytes);
119 }
120
121 return d_instance.get();
122}

124string get_hash(const string &s) {
125 if (s.empty()) {
126 string msg = "You cannot hash the empty string.";
127 BESDEBUG(HTTP_MODULE, prolog << msg << endl);
128 throw BESInternalError(msg, __FILE__, __LINE__);
129 }
130 return picosha2::hash256_hex_string(s[0] == '/' ? s : "/" + s);
131}

133bool is_url(const string &candidate) {
134 size_t index = candidate.find(HTTP_PROTOCOL);
135 if (index) {
136 index = candidate.find(HTTPS_PROTOCOL);
137 if (index) {
138 return false;
139 }
140 }
141 return true;
142}

151string get_real_name_extension(const string &identifier) {
152 string path_part;
153
154 if (is_url(identifier)) {
155 // Since it's a URL it might have a massive query string attached, and since we
156 // have no idea what the query parameters mean, we'll just punt and look at the
157 // path part of the URL. We make an instance of http::url which will carve up
158 // the URL for us.
159 http::url target_url(identifier);
160 path_part = target_url.path();
161 }
162 else {
163 path_part = identifier;
164 }
165
166 string real_name_extension;
167 vector<string> path_elements;
168 // Now that we have a "path" (none of that query string mess) we can tokenize it.
169 BESUtil::tokenize(path_part, path_elements);
170 if (!path_elements.empty()) {
171 string last = path_elements.back();
172 if (last != path_part)
173 real_name_extension = "#" + last; // This utilizes a hack in libdap
174 }
175 return real_name_extension;
176}

191string HttpCache::get_cache_file_name(const string &uid, const string &src_id, bool mangle) const {
192 string uid_part = uid.empty() ? "" : uid + "_";
193 string src_id_part = mangle ? get_hash(src_id) : src_id;
194
195 string cache_filename = get_cache_file_prefix() + uid_part + src_id_part + get_real_name_extension(src_id);
196
197 return BESUtil::assemblePath(get_cache_directory(), cache_filename);
198}
