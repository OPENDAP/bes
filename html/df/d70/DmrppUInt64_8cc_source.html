54bool
55DmrppUInt64::read()
56{
57 BESDEBUG("dmrpp", "Entering " <<__PRETTY_FUNCTION__ << " for '" << name() << "'" << endl);
58
59 if (!get_chunks_loaded())
60 load_chunks(this);
61
62 if (read_p())
63 return true;
64
65 set_value(*reinterpret_cast<dods_uint64*>(read_atomic(name())));
66
67 if ( this->twiddle_bytes() ) {
68 d_buf = bswap_64(d_buf);
69 }
70 set_read_p(true);
71
72 return true;
73
74}
