38template<typename OutIter>
39void input_hex(std::istream& is, OutIter first, OutIter last){
40 char c;
41 std::vector<char> buffer;
42 while(first != last){
43 if(buffer.size() == 2){
44 *(first++) = (buffer.front()*16+buffer.back());
45 buffer.clear();
46 }
47 is >> c;
48 if('0' <= c && c <= '9'){
49 buffer.push_back(c-'0');
50 }else
51 if('a' <= c && c <= 'f'){
52 buffer.push_back(c-'a'+10);
53 }
54 }
55}
