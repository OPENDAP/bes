68void
69FONcDouble::define(int ncid) {
70 FONcBaseType::define(ncid);
71
72 if (!d_defined) {
73
74 if (d_is_dap4) {
75 D4Attributes *d4_attrs = _f->attributes();
76 updateD4AttrType(d4_attrs, NC_DOUBLE);
77 }
78 else {
79 AttrTable &attrs = _f->get_attr_table();
80 updateAttrType(attrs, NC_DOUBLE);
81 }
82
83 FONcAttributes::add_variable_attributes(ncid, d_varid, _f, isNetCDF4_ENHANCED(), d_is_dap4);
84 FONcAttributes::add_original_name(ncid, d_varid, d_varname, d_orig_varname);
85
86 d_defined = true;
87 }

97void
98FONcDouble::write(int ncid) {
99 BESDEBUG("fonc", "FONcDouble::write for var " << d_varname << endl);
100
101 if (d_is_dap4)
102 _f->intern_data();
103 else
104 _f->intern_data(*get_eval(), *get_dds());
105
106 double data = _f->value();
107 size_t var_index[] = {0};
108 int stax = nc_put_var1_double(ncid, d_varid, var_index, &data);
109 if (stax != NC_NOERR) {
110 string err = (string) "fileout.netcdf - "
111 + "Failed to write double data for "
112 + d_varname;
113 FONcUtils::handle_error(stax, err, __FILE__, __LINE__);
114 }
115
116 BESDEBUG("fonc", "FONcDouble::done write for var " << d_varname << endl);
117}
